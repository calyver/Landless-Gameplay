can_leave_court_trigger = {
	OR = {
		NOT = {
			exists = court_owner
		}
		AND = {
			exists = court_owner
			has_any_court_position = no
			NOR = {
				any_close_family_member = {
					this = court_owner
				}
				any_consort = {
					this = court_owner
				}
			}
			is_councillor = no
			is_knight = no
		}
	}
}

can_visit_court_owner_trigger = {
	NOR = {
		has_relation_grudge = OWNER
		has_relation_rival = OWNER
		has_relation_nemesis = OWNER
	}
}
